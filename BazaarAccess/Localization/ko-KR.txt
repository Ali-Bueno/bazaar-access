# 이 파일은 영어 참조 카탈로그(en-US.txt)의 번역본입니다.
# 번역에 없는 키는 모드가 영어로 대체하므로, 부분 번역도 괜찮습니다 -
# 번역할 수 있는 것만 번역하고 나머지는 비워두세요.
#
# 형식
#   key = 텍스트
#   '#'로 시작하는 줄은 주석입니다. 빈 줄은 무시됩니다.
#   {0}, {1}... 은 모드가 채워 넣는 자리표시자입니다 (아이템 이름, 수치 등).
#   영어 텍스트에 있는 자리표시자는 모두 유지하세요 - 언어의 어순에 맞게
#   순서를 바꿔도 됩니다.
#   줄바꿈에는 \n을 사용하세요.
#
# 개수
#   .one / .other로 끝나는 키는 수량에 따라 선택됩니다: 단수는 .one,
#   그 외는 .other 입니다. 구분이 필요 없는 언어는 둘 다 같은 텍스트를 쓰면 됩니다.
#
# Shield, Burn, Bronze, Weapon 같은 단어는 여기 없습니다: 모드가 게임
# 자체에서 이미 번역된 단어를 그대로 가져오므로, 카드가 말하는 것과 정확히
# 같은 단어를 사용합니다.

# BazaarAccess - 최종 로컬라이제이션 작업 (신규 키)
#
# 아래는 DetailLineBuilder.cs에 대한 최종 잔여 항목 정리 작업에서 새로 추가된
# 키입니다. 아직 en-US.txt에 병합되지 않았으며, 관리자가 수동으로 병합합니다.
#
# 아래 네 개의 키는 모두 스탯 자체의 게임 용어 단어(Cooldown/Damage/Heal/
# Shield...)를 인자로 받습니다. 주변 형식("seconds", "s", 쉼표, 어순)만
# 모드가 직접 구성한 것입니다.

# --- Gameplay/CardReading/DetailLineBuilder.cs ---

# 독립된 상세 줄: "Cooldown 5.0 seconds" ({0} = 라벨, {1} = 값)
card.stat.cooldownFull = {0} {1}초

# 더 긴 문장 안의 삽입구: ", Cooldown 5.0s" ({0} = 라벨, {1} = 값)
card.stat.cooldownShort = , {0} {1}초

# 라벨 없는 값 + 단위 (적 요약용): "5.0s"
card.stat.secondsBare = {0}초

# 아이템 크기에 대한 삽입구: ", Size 2" ({0} = 슬롯 수)
card.stat.sizeInline = , 크기 {0}

# 라벨 없는 간결한 "값 라벨" 스탯 (예: "180 damage"). card.stat.inline과
# 인자 순서가 반대입니다 ({0} = 값, {1} = 라벨) - 언어마다 자연스러운 어순으로
# 쓸 수 있도록 별도 키로 분리했습니다.
card.stat.compact = {0} {1}
